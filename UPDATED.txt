interface ButtonProps {
  variant?: 'primary' | 'secondary' | 'success' | 'danger'
  size?: 'sm' | 'md' | 'lg'
  // ... other props
}

export default function Button({ variant = 'primary', size = 'md', ... }: ButtonProps) {
  const variants = {
    primary: 'bg-gradient-to-r from-slate-600 to-slate-700 hover:from-slate-700 hover:to-slate-800 text-white',
    secondary: 'bg-white/60 hover:bg-white/80 text-gray-700',
    success: 'bg-gradient-to-r from-green-600 to-green-700 hover:from-green-700 hover:to-green-800 text-white',
    danger: 'bg-gradient-to-r from-red-600 to-red-700 hover:from-red-700 hover:to-red-800 text-white',
  }
  
  const sizes = {
    sm: 'px-3 py-2 text-sm',
    md: 'px-4 py-3 text-base',  // ‚úÖ py-3 for tablets!
    lg: 'px-6 py-4 text-lg',
  }
  
  return (
    <button className={`${variants[variant]} ${sizes[size]} rounded-lg font-semibold transition-base hover:scale-[1.02] shadow-lg`}>
      {children}
    </button>
  )
}
